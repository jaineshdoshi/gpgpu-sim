72 int BatchTrafficManager::_IssuePacket( int source, int cl )
73 {
74  int result = 0;
75  if(_use_read_write[cl]) { //read write packets
76  //check queue for waiting replies.
77  //check to make sure it is on time yet
78  if(!_repliesPending[source].empty()) {
79  if(_repliesPending[source].front()->time <= _time) {
80  result = -1;
81  }
82  } else {
83  if((_packet_seq_no[source] < _batch_size) &&
84  ((_max_outstanding <= 0) ||
85  (_requestsOutstanding[source] < _max_outstanding))) {
86 
87  //coin toss to determine request type.
88  result = (RandomFloat() < 0.5) ? 2 : 1;
89 
90  _requestsOutstanding[source]++;
91  }
92  }
93  } else { //normal
94  if((_packet_seq_no[source] < _batch_size) &&
95  ((_max_outstanding <= 0) ||
96  (_requestsOutstanding[source] < _max_outstanding))) {
97  result = _GetNextPacketSize(cl);
98  _requestsOutstanding[source]++;
99  }
100  }
101  if(result != 0) {
102  _packet_seq_no[source]++;
103  }
104  return result;
105 }

113 bool BatchTrafficManager::_SingleSim( )
114 {
115  int batch_index = 0;

117  _packet_seq_no.assign(_nodes, 0);

142  int empty_steps = 0;
143 
144  bool packets_left = false;
145  for(int c = 0; c < _classes; ++c) {
146  packets_left |= !_total_in_flight_flits[c].empty();
147  }

152  ++empty_steps;

155  _DisplayRemaining( );
156  cout << ".";
157  }
158

161  packets_left |= !_total_in_flight_flits[c].empty();
162  }
163  }

166  << "Last packet was " << _last_pid << ", last flit was " << _last_id << "." << endl;

172  UpdateStats();
