80 void VC::AddFlit( Flit *f )
81 {
82  assert(f);
83 
84  if(_expected_pid >= 0) {
85  if(f->pid != _expected_pid) {
86  ostringstream err;
87  err << "Received flit " << f->id << " with unexpected packet ID: " << f->pid
88  << " (expected: " << _expected_pid << ")";
89  Error(err.str());
90  } else if(f->tail) {
91  _expected_pid = -1;
92  }
93  } else if(!f->tail) {
94  _expected_pid = f->pid;
95  }
96 
97  // update flit priority before adding to VC buffer
98  if(_pri_type == local_age_based) {
99  f->pri = numeric_limits<int>::max() - GetSimTime();
100  assert(f->pri >= 0);
101  } else if(_pri_type == hop_count_based) {
102  f->pri = f->hops;
103  assert(f->pri >= 0);
104  }
105 
106  _buffer.push_back(f);
107  UpdatePriority();
108 }

157 void VC::UpdatePriority()
158 {
159  if(_buffer.empty()) return;
160  if(_pri_type == queue_length_based) {
161  _pri = _buffer.size();
162  } else if(_pri_type != none) {
163  Flit * f = _buffer.front();
164  if((_pri_type != local_age_based) && _priority_donation) {
165  Flit * df = f;
166  for(size_t i = 1; i < _buffer.size(); ++i) {
167  Flit * bf = _buffer[i];
168  if(bf->pri > df->pri) df = bf;
169  }
170  if((df != f) && (df->watch || f->watch)) {
171  *gWatchOut << GetSimTime() << " | " << FullName() << " | "
172  << "Flit " << df->id
173  << " donates priority to flit " << f->id
174  << "." << endl;
175  }
176  f = df;
177  }
178  if(f->watch)
179  *gWatchOut << GetSimTime() << " | " << FullName() << " | "
180  << "Flit " << f->id
181  << " sets priority to " << f->pri
182  << "." << endl;
183  _pri = f->pri;
184  }
185 }
