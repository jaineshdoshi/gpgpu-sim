426 Allocator *Allocator::NewAllocator( Module *parent, const string& name,
427  const string &alloc_type,
428  int inputs, int outputs,
429  Configuration const * const config )
430 {
431  Allocator *a = 0;
432 
433  string alloc_name;
434  string param_str;
435  size_t left = alloc_type.find_first_of('(');
436  if(left == string::npos) {
437  alloc_name = alloc_type;
438  } else {
439  alloc_name = alloc_type.substr(0, left);
440  size_t right = alloc_type.find_last_of(')');
441  if(right == string::npos) {
442  param_str = alloc_type.substr(left+1);
443  } else {
444  param_str = alloc_type.substr(left+1, right-left-1);
445  }
446  }
447  if ( alloc_name == "max_size" ) {
448  a = new MaxSizeMatch( parent, name, inputs, outputs );
449  } else if ( alloc_name == "pim" ) {
450  int iters = param_str.empty() ? (config ? config->GetInt("alloc_iters") : 1) : atoi(param_str.c_str());
451  a = new PIM( parent, name, inputs, outputs, iters );
452  } else if ( alloc_name == "islip" ) {
453  int iters = param_str.empty() ? (config ? config->GetInt("alloc_iters") : 1) : atoi(param_str.c_str());
454  a = new iSLIP_Sparse( parent, name, inputs, outputs, iters );
455  } else if ( alloc_name == "loa" ) {
456  a = new LOA( parent, name, inputs, outputs );
457  } else if ( alloc_name == "wavefront" ) {
458  a = new Wavefront( parent, name, inputs, outputs );
459  } else if ( alloc_name == "rr_wavefront" ) {
460  a = new Wavefront( parent, name, inputs, outputs, true );
461  } else if ( alloc_name == "select" ) {
462  int iters = param_str.empty() ? (config ? config->GetInt("alloc_iters") : 1) : atoi(param_str.c_str());
463  a = new SelAlloc( parent, name, inputs, outputs, iters );
464  } else if (alloc_name == "separable_input_first") {
465  string arb_type = param_str.empty() ? (config ? config->GetStr("arb_type") : "round_robin") : param_str;
466  a = new SeparableInputFirstAllocator( parent, name, inputs, outputs,
467  arb_type );
468  } else if (alloc_name == "separable_output_first") {
469  string arb_type = param_str.empty() ? (config ? config->GetStr("arb_type") : "round_robin") : param_str;
470  a = new SeparableOutputFirstAllocator( parent, name, inputs, outputs,
471  arb_type );
472  }
473 
474 //==================================================
475 // Insert new allocators here, add another else if
476 //==================================================
477 
478 
479  return a;
480 }
