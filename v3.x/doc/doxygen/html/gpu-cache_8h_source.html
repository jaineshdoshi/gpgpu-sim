138  void init(char * config, FuncCache status)
139  {
140  cache_status= status;
141  assert( config );
142  char rp, wp, ap, mshr_type, wap;
143 
144  int ntok = sscanf(config,"%u:%u:%u,%c:%c:%c:%c,%c:%u:%u,%u:%u,%u",
145  &m_nset, &m_line_sz, &m_assoc, &rp, &wp, &ap, &wap,
146  &mshr_type, &m_mshr_entries,&m_mshr_max_merge,
147  &m_miss_queue_size,&m_result_fifo_entries,
148  &m_data_port_width);
149 
150  if ( ntok < 11 ) {
151  if ( !strcmp(config,"none") ) {
152  m_disabled = true;
153  return;
154  }
155  exit_parse_error();
156  }
157  switch (rp) {
158  case 'L': m_replacement_policy = LRU; break;
159  case 'F': m_replacement_policy = FIFO; break;
160  default: exit_parse_error();
161  }
162  switch (wp) {
163  case 'R': m_write_policy = READ_ONLY; break;
164  case 'B': m_write_policy = WRITE_BACK; break;
165  case 'T': m_write_policy = WRITE_THROUGH; break;
166  case 'E': m_write_policy = WRITE_EVICT; break;
167  case 'L': m_write_policy = LOCAL_WB_GLOBAL_WT; break;
168  default: exit_parse_error();
169  }
170  switch (ap) {
171  case 'm': m_alloc_policy = ON_MISS; break;
172  case 'f': m_alloc_policy = ON_FILL; break;
173  default: exit_parse_error();
174  }
175  switch (mshr_type) {
176  case 'F': m_mshr_type = TEX_FIFO; assert(ntok==12); break;
177  case 'A': m_mshr_type = ASSOC; break;
178  default: exit_parse_error();
179  }
180  m_line_sz_log2 = LOGB2(m_line_sz);
181  m_nset_log2 = LOGB2(m_nset);
182  m_valid = true;
183 
184  switch(wap){
185  case 'W': m_write_alloc_policy = WRITE_ALLOCATE; break;
186  case 'N': m_write_alloc_policy = NO_WRITE_ALLOCATE; break;
187  default: exit_parse_error();
188  }
189 
190  // detect invalid configuration
191  if (m_alloc_policy == ON_FILL and m_write_policy == WRITE_BACK) {
192  // A writeback cache with allocate-on-fill policy will inevitably lead to deadlock:
193  // The deadlock happens when an incoming cache-fill evicts a dirty
194  // line, generating a writeback request. If the memory subsystem
195  // is congested, the interconnection network may not have
196  // sufficient buffer for the writeback request. This stalls the
197  // incoming cache-fill. The stall may propagate through the memory
198  // subsystem back to the output port of the same core, creating a
199  // deadlock where the wrtieback request and the incoming cache-fill
200  // are stalling each other.
201  assert(0 && "Invalid cache configuration: Writeback cache cannot allocate new line on fill. ");
202  }
203 
204  // default: port to data array width and granularity = line size
205  if (m_data_port_width == 0) {
206  m_data_port_width = m_line_sz;
207  }
208  assert(m_line_sz % m_data_port_width == 0);
209  }
